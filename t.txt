    játékos kap egy lapot a (véletlenszerü függvénytől)                 # a lap eltávolítása a pakkliból
    gép kap egy lapot a (véletlenszerü függvénytől)                     # a lap eltávolítása a pakkliból
    játékos kap még egy lapot a (véletlenszerü függvénytől)             # a lap eltávolítása a pakkliból
    gép kap még egy lapot a (véletlenszerü függvénytől)                 # a lap eltávolítása a pakkliból
    return játékos lapjainak összege
    return gép lapjainak összege
